# WebPack

module.export = export default 랑 같은 개념

  • 기본 webpack 설정
var path = require("path”); // nodejs에서 제공되는 개념(http://chromeextension.kr/2016/08/node-js-path/)

module.exports = {
  mode: "none",  // 모드 (none : 없음 , dev : 개발, production : 배포)
  entry: "./src/index.js", //시작 지점 설정
  output: { //생성되는 경로
    filename: "main.js", //아웃풋 파일 명
    path: path.resolve(__dirname, "dist"), //아웃풋 폴더
  },
};

  • js가 아닌 파일을 로드하여 구분 (ex : css)
var path = require("path");
var MiniCssExtractPlugin = require("mini-css-extract-plugin");

module.exports = {
  mode: "none",
  //변환할 파일 정보 ( 시작파일 )
  entry: "./app/index.js",
  // 변환된 파일의 정보 ( 결과 파일)
  output: {
    filename: "bundle.js",
    path: path.resolve(__dirname, "dist"),
  },
  // 변환하면서 적용할 도구들 ( 비 자바스크립트 파일에 대해서 처리 )
  module: {
    rules: [
      {
        test: /\.css$/, // .css 파일 관련 처리
        use: ["style-loader", "css-loader"], //css 파일관련해서  style-loader와 css-loader를 쓰겠다
        // use 순서 중요 (오른쪽 부터 왼쪽으로 적용된다)
        // style-loader 스타일 태그 로더
        // css-loader  css 파일 로더
      },
    ],
  },
  //변환된 결과에 대해서 추가적인 동작, 기능 수행
  plugins: [new MiniCssExtractPlugin()],
};
